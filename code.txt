function showReadyMessage() {
  SpreadsheetApp.getUi().alert('AI評価メニューの準備ができています。');
}

function saveGeminiApiKey() {
  const ui = SpreadsheetApp.getUi();
  const response = ui.prompt(
    'Gemini APIキー登録',
    'GeminiのAPIキーを入力してください',
    ui.ButtonSet.OK_CANCEL
  );

  if (response.getSelectedButton() !== ui.Button.OK) {
    ui.alert('キャンセルしました。');
    return;
  }

  const apiKey = response.getResponseText().trim();

  if (!apiKey) {
    ui.alert('APIキーが空です。');
    return;
  }

  PropertiesService.getScriptProperties().setProperty('GEMINI_API_KEY', apiKey);
  ui.alert('Gemini APIキーを保存しました。');
}

  if (!apiKey) {

    return;
  }

  );
}

function testGeminiConnection() {
  const apiKey = PropertiesService.getScriptProperties().getProperty('GEMINI_API_KEY');

  if (!apiKey) {
    SpreadsheetApp.getUi().alert('Gemini APIキーが保存されていません。');
    return;
  }

  const url = 'https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key=' + apiKey;

  const payload = {
    contents: [
      {
        parts: [
          { text: '「接続テスト成功です」とだけ日本語で返してください。' }
        ]
      }
    ]
  };

  const options = {
    method: 'post',
    contentType: 'application/json',
    payload: JSON.stringify(payload),
    muteHttpExceptions: true
  };

  const response = UrlFetchApp.fetch(url, options);
  const statusCode = response.getResponseCode();
  const bodyText = response.getContentText();

  if (statusCode !== 200) {
    SpreadsheetApp.getUi().alert('Gemini接続エラー\nHTTP ' + statusCode + '\n' + bodyText);
    return;
  }

  const json = JSON.parse(bodyText);
  let resultText = '';

  try {
    resultText = json.candidates[0].content.parts[0].text;
  } catch (e) {
    SpreadsheetApp.getUi().alert('レスポンス解析に失敗しました。\n' + bodyText);
    return;
  }

  SpreadsheetApp.getUi().alert('Gemini接続成功\n\n返答: ' + resultText);
}

function scoreSecondRowWithGeminiAndAggregate() {
  const scoreResult = scoreMissionRowWithGemini(2);

  if (!scoreResult.success) {
    SpreadsheetApp.getUi().alert('AI採点エラー\n' + scoreResult.error);
    return;
  }

  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('missions');
  const key = buildAggregationKeyFromRow(sheet.getRange(2, 1, 1, 28).getValues()[0]);
  const aggregateResult = aggregatePersonByKey(key);

  if (!aggregateResult.success) {
    SpreadsheetApp.getUi().alert('集計エラー\n' + aggregateResult.error);
    return;
  }

  SpreadsheetApp.getUi().alert(
    '2行目のAI採点＋集計が完了しました。\n\n' +
    '対象キー: ' + key + '\n' +
    '最終総合ポイント: ' + aggregateResult.totalPoint + '\n' +
    '最終グレード: ' + aggregateResult.finalGradeLabel + '\n' +
    '最終グレード給: ' + aggregateResult.finalGradePay
  );
}

function scoreAllRowsWithGemini() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('missions');

  if (!sheet) {
    SpreadsheetApp.getUi().alert('missions シートが見つかりません。');
    return;
  }

  const lastRow = sheet.getLastRow();
  if (lastRow < 2) {
    SpreadsheetApp.getUi().alert('データがありません。');
    return;
  }

  let successCount = 0;
  let errorCount = 0;

  for (let rowNumber = 2; rowNumber <= lastRow; rowNumber++) {
    const row = sheet.getRange(rowNumber, 1, 1, 28).getValues()[0];
    const missionName = String(row[7] || '').trim();
    if (!missionName) continue;

    const result = scoreMissionRowWithGemini(rowNumber);
    if (result.success) {
      successCount++;
    } else {
      errorCount++;
    }

    Utilities.sleep(300);
  }

  SpreadsheetApp.getUi().alert(
    '全行のAI採点が完了しました。\n成功: ' + successCount + '\n失敗: ' + errorCount
  );
}

function aggregateAllPeople() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('missions');

  if (!sheet) {
    SpreadsheetApp.getUi().alert('missions シートが見つかりません。');
    return;
  }

  const lastRow = sheet.getLastRow();
  if (lastRow < 2) {
    SpreadsheetApp.getUi().alert('データがありません。');
    return;
  }

  const values = sheet.getRange(2, 1, lastRow - 1, 28).getValues();
  const uniqueKeys = {};

  values.forEach(function(row) {
    const missionName = String(row[7] || '').trim();
    if (!missionName) return;

    const key = buildAggregationKeyFromRow(row);
    uniqueKeys[key] = true;
  });

  const keys = Object.keys(uniqueKeys);
  let successCount = 0;
  let errorCount = 0;

  keys.forEach(function(key) {
    const result = aggregatePersonByKey(key);
    if (result.success) {
      successCount++;
    } else {
      errorCount++;
    }
  });

  SpreadsheetApp.getUi().alert(
    '全体集計が完了しました。\n対象人数: ' + keys.length + '\n成功: ' + successCount + '\n失敗: ' + errorCount
  );
}

function scoreMissionRowWithGemini(rowNumber) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('missions');

  if (!sheet) {
    return { success: false, error: 'missions シートが見つかりません。' };
  }

  const row = sheet.getRange(rowNumber, 1, 1, 28).getValues()[0];
  const missionName = String(row[7] || '').trim();

  if (!missionName) {
    return { success: false, error: rowNumber + '行目のM1_ミッション名（H列）が空です。' };
  }

  const inputData = {
    period: row[0],
    name: row[1],
    department: row[2],
    currentGrade: row[3],
    position: row[4],
    missionNo: row[5],
    weight: row[6],
    m1_missionName: row[7],
    m2_backgroundGoal: row[8],
    m3_contentDifficulty: row[9],
    m4_stakeholdersRole: row[10],
    m5_feasibilityEvidence: row[11]
  };

  const prompt = buildScoringPrompt(inputData);
  const result = callGeminiForScoring(prompt);

  if (!result.success) {
    sheet.getRange(rowNumber, 24).setValue('AIエラー');
    return result;
  }

  const parsed = result.data;
  const missionWeightedPoint = calculateMissionWeightedPoint(parsed, Number(row[6] || 0));

  sheet.getRange(rowNumber, 13).setValue(parsed.difficulty);    // M
  sheet.getRange(rowNumber, 14).setValue(parsed.scope);         // N
  sheet.getRange(rowNumber, 15).setValue(parsed.innovation);    // O
  sheet.getRange(rowNumber, 16).setValue(parsed.contribution);  // P
  sheet.getRange(rowNumber, 17).setValue(parsed.roleLevel);     // Q
  sheet.getRange(rowNumber, 18).setValue(parsed.feasibility);   // R
  sheet.getRange(rowNumber, 19).setValue(parsed.comment);       // S
  sheet.getRange(rowNumber, 20).setValue(missionWeightedPoint); // T
  sheet.getRange(rowNumber, 23).setValue(new Date());           // W
  sheet.getRange(rowNumber, 24).setValue('AI採点完了');         // X
  sheet.getRange(rowNumber, 25).setValue(buildAggregationKeyFromRow(row)); // Y

  return { success: true };
}

function calculateMissionWeightedPoint(parsed, weight) {
  const basePoint = roundToTwo(
    Number(parsed.difficulty) * 0.20 +
    Number(parsed.scope) * 0.15 +
    Number(parsed.innovation) * 0.20 +
    Number(parsed.contribution) * 0.20 +
    Number(parsed.roleLevel) * 0.15 +
    Number(parsed.feasibility) * 0.10
  );

  return roundToTwo(basePoint * (weight / 100));
}

function aggregatePersonByKey(targetKey) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('missions');

  if (!sheet) {
    return { success: false, error: 'missions シートが見つかりません。' };
  }

  const lastRow = sheet.getLastRow();
  if (lastRow < 2) {
    return { success: false, error: 'データがありません。' };
  }

  const values = sheet.getRange(2, 1, lastRow - 1, 28).getValues();
  const matchedRows = [];

  values.forEach(function(row, index) {
    const key = buildAggregationKeyFromRow(row);
    const missionName = String(row[7] || '').trim();
    if (!missionName) return;

    if (key === targetKey) {
      matchedRows.push({
        rowNumber: index + 2,
        row: row
      });
    }
  });

  if (matchedRows.length === 0) {
    return { success: false, error: '対象データが見つかりません。' };
  }

  let totalPoint = 0;
  let position = '';

  for (let i = 0; i < matchedRows.length; i++) {
    const row = matchedRows[i].row;
    position = String(row[4] || '').trim();
    const missionWeightedPoint = Number(row[19] || 0);

    if (!missionWeightedPoint) {
      return { success: false, error: '未採点のミッションがあります。先にAI採点を実行してください。' };
    }

    totalPoint += missionWeightedPoint;
  }

  totalPoint = roundToTwo(totalPoint);

  if (!position) {
    return { success: false, error: '役職が空です。' };
  }

  const finalGradeNumber = convertPointToGradeByPosition(position, totalPoint);
  const finalGradeLabel = convertGradeNumberToLabel(position, finalGradeNumber);
  const finalGradePay = getGradePay(position, finalGradeNumber);

  matchedRows.forEach(function(item) {
    const rowNumber = item.rowNumber;
    sheet.getRange(rowNumber, 21).setValue(finalGradeLabel);   // U
    sheet.getRange(rowNumber, 22).setValue(finalGradePay);     // V
    sheet.getRange(rowNumber, 24).setValue('集計完了');         // X
    sheet.getRange(rowNumber, 25).setValue(targetKey);         // Y
    sheet.getRange(rowNumber, 26).setValue(totalPoint);        // Z
    sheet.getRange(rowNumber, 27).setValue(finalGradeLabel);   // AA
    sheet.getRange(rowNumber, 28).setValue(finalGradePay);     // AB
  });

  return {
    success: true,
    totalPoint: totalPoint,
    finalGradeNumber: finalGradeNumber,
    finalGradeLabel: finalGradeLabel,
    finalGradePay: finalGradePay
  };
}

function buildAggregationKeyFromRow(row) {
  const manualKey = String(row[24] || '').trim();
  if (manualKey) return manualKey;

  const period = String(row[0] || '').trim();
  const name = String(row[1] || '').trim();
  const position = String(row[4] || '').trim();

  return period + '_' + name + '_' + position;
}

function normalizePositionGroup(position) {
  const p = String(position || '').trim();

  if (['担当', 'チーフ', '支店長代理'].includes(p)) {
    return 'groupA';
  }

  if (['支店長', '課長'].includes(p)) {
    return 'groupL';
  }

  if (['支社長', 'ユニットマネージャー'].includes(p)) {
    return 'groupU';
  }

  if (['統括', 'グループマネージャー'].includes(p)) {
    return 'groupG';
  }

  if (['本部長', 'ディビジョンマネージャー'].includes(p)) {
    return 'groupD';
  }

  return 'groupA';
}

function getGradePrefixByPosition(position) {
  const group = normalizePositionGroup(position);

  const prefixMap = {
    groupA: 'A',
    groupL: 'L',
    groupU: 'U',
    groupG: 'G',
    groupD: 'D'
  };

  return prefixMap[group] || 'A';
}

function convertGradeNumberToLabel(position, gradeNumber) {
  return getGradePrefixByPosition(position) + String(gradeNumber);
}

function convertPointToGradeByPosition(position, totalPoint) {
  const thresholds = getThresholdsByPosition(position);

  for (let i = 0; i < thresholds.length; i++) {
    if (totalPoint >= thresholds[i].minPoint) {
      return thresholds[i].grade;
    }
  }

  return 1;
}

function callGeminiForScoring(prompt) {
  const apiKey = PropertiesService.getScriptProperties().getProperty('GEMINI_API_KEY');

  if (!apiKey) {
    return { success: false, error: 'Gemini APIキーが保存されていません。' };
  }

  const url = 'https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key=' + apiKey;

  const payload = {
    contents: [
      {
        parts: [
          { text: prompt }
        ]
      }
    ]
  };

  const options = {
    method: 'post',
    contentType: 'application/json',
    payload: JSON.stringify(payload),
    muteHttpExceptions: true
  };

  const response = UrlFetchApp.fetch(url, options);
  const statusCode = response.getResponseCode();
  const bodyText = response.getContentText();

  if (statusCode !== 200) {
    return { success: false, error: 'HTTP ' + statusCode + '\n' + bodyText };
  }

  try {
    const json = JSON.parse(bodyText);
    const text = json.candidates[0].content.parts[0].text;
    const cleaned = cleanJsonText(text);
    const parsed = JSON.parse(cleaned);

    return { success: true, data: parsed };
  } catch (e) {
    return { success: false, error: 'レスポンス解析失敗\n' + bodyText };
  }
}

function cleanJsonText(text) {
  return String(text || '')
    .replace(/```json/g, '')
    .replace(/```/g, '')
    .trim();
}

function roundToTwo(value) {
  return Math.round(value * 100) / 100;
}
